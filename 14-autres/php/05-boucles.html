Créer une boucle en PHP

Comme en JS, while / for

<?php 
	$inc = 1;
	while($inc <= 10){
		echo $inc;

		$inc++;
	}

	for($i=0; $i <= 10; $i++){
		echo $inc;
	}
?>

Foreach

foreach permet de lister le contenu de tout un tableau

<?php 
	$tab = ["fraise", "banane", "courge"];
	
	foreach($tab as $item){ // pour tous les éléments du tableau $tab, je stock l'élément courant dans $item
		echo $item; // item correspond à l'élément courant
	}
?>

Foreach et tableaux associatifs

<?php 
	$tab = [
		"fruit" => "fraise",
		"legume" => "courge",
		"boisson" => "eau"
	];
	
	foreach($tab as $key=>$item){ // pour tous les éléments du tableau $tab, je stock l'élément courant dans $item
		echo $key." - ".$item; // item correspond à l'élément courant / key correspond à la clé
	}
?>
